### Steps to Reproduce

Following instructions for [building artifacts](https://github.com/ROCm/TheRock?tab=readme-ov-file#building-from-source) and [building package from artifacts](https://github.com/ROCm/TheRock/blob/main/docs/packaging/python_packaging.md#building-packages).
```
# Install Ubuntu dependencies
sudo apt update
sudo apt install gfortran git ninja-build cmake g++ pkg-config xxd patchelf automake libtool python3-venv python3-dev libegl1-mesa-dev

# Clone the repository
git clone https://github.com/ROCm/TheRock.git
cd TheRock

# Init python virtual environment and install python dependencies
python3 -m venv .venv && source .venv/bin/activate
pip install --upgrade pip
pip install -r requirements.txt
pip install piprepo setuptools

# Download submodules and apply patches
python3 ./build_tools/fetch_sources.py

eval "$(./build_tools/setup_ccache.py)"
cmake -B build -GNinja -DTHEROCK_AMDGPU_TARGETS="gfx906" \
  -DCMAKE_C_COMPILER_LAUNCHER=ccache \
  -DCMAKE_CXX_COMPILER_LAUNCHER=ccache \
  .

cmake --build build

./build_tools/build_python_packages.py \
    --artifact-dir ./build/artifacts \
    --dest-dir ${HOME}/tmp/packages \
    --version 7.0.0.dev0

piprepo build ${HOME}/tmp/packages/dist
pip install rocm[libraries,devel]==7.0.0.dev0 \
  --extra-index-url ${HOME}/tmp/packages/dist/simple \
  --force-reinstall --no-cache-dir

pip3 install --pre torch torchvision --index-url https://download.pytorch.org/whl/nightly/rocm7.0
```
