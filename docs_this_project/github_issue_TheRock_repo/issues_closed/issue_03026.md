### Steps to Reproduce

This script shows float32 and float16 comparison. 
```
import torch
from torch import nn
import numpy as np

def verify_tensor(expected, result, atol=None, rtol=None):
    if atol is None:
        atol = 1e-2
    if rtol is None:
        rtol = 1e-3
    passed = np.allclose(expected, result, rtol=rtol, atol=atol)
    if passed:
        print("Tensors are the same")
        return True

    abs_diff = np.abs(expected - result)
    bool_tensor = np.isclose(result, expected, rtol=rtol, atol=atol)
    where_error = np.argwhere(~bool_tensor)
    max_error = np.argwhere(abs_diff == abs_diff.max())

    if not len(where_error):
        print("Tensors are the same")
        return True
    percentage_error = (len(where_error)*100) / float(np.prod(result.shape))
    message = f'Abs error (max: {abs_diff.max()}, min: {abs_diff.min()}), num_errors: {len(where_error)}, {percentage_error:.1f}%..'
    print(message)
    return False


if not torch.cuda.is_available():
    print("!!! gpu not available !!!")
    exit

gpu_device = torch.device('cuda')
cpu_device = torch.device('cpu')

with torch.no_grad():
    conv_in = torch.randn((1, 32, 135, 240), dtype=torch.float32)

    conv_with_groups = nn.Conv2d(
        32,
        32,
        kernel_size=3,
        padding=1,
        groups=2,
        bias=True,
        padding_mode="zeros",
    )
    def conv_inference(conv_input, conv_op):
        conv_input = conv_input.to(cpu_device)
        conv_op = conv_op.to(cpu_device)
        conv_out = conv_op(conv_input)
        
        conv_in_gpu = conv_input.to(gpu_device)
        conv_op_gpu = conv_op.to(gpu_device)
        conv_out_gpu = conv_op_gpu(conv_in_gpu)

        verify_tensor(conv_out, conv_out_gpu.to(cpu_device))


    print("Float32 comparison")
    conv_inference(conv_in, conv_with_groups)

    print("Float16 comparison")
    conv_in_fp16 = conv_in.to(torch.float16)
    conv_with_groups_fp16 = conv_with_groups.half()
    conv_inference(conv_in_fp16, conv_with_groups_fp16)

    
```
