### Problem Description

pytorch is utilising GTT instead VRAM on AMD RYZEN AI MAX+ PRO 395 w/ Radeon 8060S

this is continuation of conversation happened on https://github.com/ROCm/TheRock/issues/1668

I have run provided script
```
import time
import torch

CARD = 0 
VRAM_PATH = f"/sys/class/drm/card{CARD}/device/mem_info_vram_used"
GTT_PATH  = f"/sys/class/drm/card{CARD}/device/mem_info_gtt_used"

def read_bytes(path):
    try:
        with open(path, "r") as f:
            return int(f.read().strip())
    except Exception:
        return None

def to_mb(b):
    return None if b is None else b / (1024 * 1024)

def bytes_to_gb(n):
    return n / (1024 ** 3)

def main(target_gb=3.0, dtype=torch.float32, safety_factor=0.98, device_index=0):
    if not torch.cuda.is_available():
        print("CUDA is not available. This will run on CPU and won't use GPU VRAM.")
        return
    device = torch.device(f"cuda:{device_index}")
    print(f"Using device: {device} ({torch.cuda.get_device_name(device_index)})")
    total = torch.cuda.get_device_properties(device).total_memory
    print(f"Device total memory reported by torch: {bytes_to_gb(total):.2f} GB ({total:,} bytes)")
    vram0 = read_bytes(VRAM_PATH)
    gtt0  = read_bytes(GTT_PATH)
    print(f"Baseline: VRAM={to_mb(vram0):.1f} MB  GTT={to_mb(gtt0):.1f} MB")
    bytes_per_elem = torch.tensor([], dtype=dtype).element_size()
    target_bytes = int(target_gb * (1024 ** 3) * safety_factor)
    num_elems = target_bytes // bytes_per_elem
    print(f"Target allocation: ~{target_gb:.2f} GB ({target_bytes:,} bytes) with dtype={dtype} -> elements: {num_elems:,}")
    t = torch.empty(num_elems, dtype=dtype, device=device)
    t.fill_(1.0)
    torch.cuda.synchronize()
    allocated = torch.cuda.memory_allocated(device)
    reserved  = torch.cuda.memory_reserved(device)
    print(f"CUDA memory allocated: {bytes_to_gb(allocated):.2f} GB ({allocated:,} bytes)")
    print(f"CUDA memory reserved:  {bytes_to_gb(reserved):.2f} GB ({reserved:,} bytes)")
    vram1 = read_bytes(VRAM_PATH)
    gtt1  = read_bytes(GTT_PATH)
    print(f"After alloc: VRAM={to_mb(vram1):.1f} MB  GTT={to_mb(gtt1):.1f} MB")
    if vram0 is not None and vram1 is not None:
        print(f"ΔVRAM: {to_mb(vram1 - vram0):.1f} MB")
    if gtt0 is not None and gtt1 is not None:
        print(f"ΔGTT:  {to_mb(gtt1 - gtt0):.1f} MB")
    print("Sleeping for 20 seconds... monitor counters in another terminal")
    time.sleep(20)
    del t
    torch.cuda.empty_cache()
    torch.cuda.synchronize()

    vram2 = read_bytes(VRAM_PATH)
    gtt2  = read_bytes(GTT_PATH)
    print(f"After free: VRAM={to_mb(vram2):.1f} MB  GTT={to_mb(gtt2):.1f} MB")
    if vram1 is not None and vram2 is not None:
        print(f"ΔVRAM (free): {to_mb(vram2 - vram1):.1f} MB")
    if gtt1 is not None and gtt2 is not None:
        print(f"ΔGTT  (free): {to_mb(gtt2 - gtt1):.1f} MB")

if __name__ == "__main__":
    main(target_gb=3.0, dtype=torch.float32, safety_factor=0.98, device_index=0)
```

for some reason on my system it is card no 1, there is no card 0
I can confirm provided script does utilize GTT, installed 7.10 version from amd rocm site
I am using ubuntu oem kernel 6.14.0-1017-oem|
