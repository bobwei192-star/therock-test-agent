Step 2. Use following code for test

Get `latents.pt` file : [latents.zip](https://github.com/user-attachments/files/23875195/latents.zip)

```
import torch
import time
from diffusers.models.autoencoders import AutoencoderKL
from diffusers.image_processor import VaeImageProcessor

vae = AutoencoderKL.from_pretrained("stabilityai/stable-diffusion-3-medium-diffusers", subfolder="vae", torch_dtype=torch.float16)
vae = vae.to("cuda")
vae_scale_factor = 2 ** (len(vae.config.block_out_channels) - 1)
image_processor  = VaeImageProcessor(vae_scale_factor=vae_scale_factor)
latents          = torch.load("latents.pt", map_location='cuda:0').to(torch.float16)

for i in range(10):
    with torch.no_grad():
        t0    = time.perf_counter()
        image = vae.decode(latents, return_dict=False)[0]
        torch.cuda.synchronize()
        print(f"Inference time {(time.perf_counter() - t0):.2f}")
        image = image_processor.postprocess(image, output_type="pil")
        image[0].save(f"text2img_{i}.png")        

print("finish")
```
