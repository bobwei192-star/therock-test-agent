#!/bin/bash
# --- STRIX HALO | TheRock 7.11 (Anti-Fragmentation Config) ---

# 1. Cleanup (Kill everything to clear the "Swiss Cheese" memory)
pkill -f "ComfyUI/main.py"
docker stop ollama 2>/dev/null

# 2. CLEAR TUNING GHOSTS
unset PYTORCH_TUNABLEOP_ENABLED
unset PYTORCH_TUNABLEOP_TUNING
unset PYTORCH_HIP_ALLOC_CONF
unset PYTORCH_ALLOC_CONF

# 3. Environment
cd /home/amd/ComfyUI
source /home/amd/ComfyUI/venv/bin/activate

# 4. ROCm Paths
export HIP_VISIBLE_DEVICES=0
export ROCM_PATH=/opt/rocm
export PATH=$PATH:/opt/rocm/bin
export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/opt/rocm/lib:/usr/lib/x86_64-linux-gnu

# 5. Strix Halo Hardware Flags
export HSA_OVERRIDE_GFX_VERSION=11.5.1
export HSA_ENABLE_SDMA=0
export FLASH_ATTENTION_TRITON_AMD_ENABLE="TRUE"

# 6. MEMORY: The Anti-Fragmentation Fix
# garbage_collection_threshold:0.6 -> Cleans up memory sooner, preventing holes.
# expandable_segments:True -> Allows PyTorch to grab new blocks if old ones are messy.
export PYTORCH_ALLOC_CONF=expandable_segments:True,garbage_collection_threshold:0.6
export PYTHONMALLOC=malloc

# 7. Launch
# --no-mmap: PREVENTS THE SLOWDOWN. Forces models into RAM, stops OS interference.
echo "🚀 Launching Strix Halo (Stable Memory Mode)..."
python main.py \
    --listen 0.0.0.0 \
    --port 8189 \
    --gpu-only \
    --disable-smart-memory \
    --disable-mmap
