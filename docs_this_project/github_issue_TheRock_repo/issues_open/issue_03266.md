### Problem Description

Even after MIOpen cache generation, AMD exhibits a persistent "First-Run Penalty" (3.01s) before stabilizing at a high-speed throughput (0.91s). In contrast, NVIDIA maintains a highly consistent profile (approx. 2s per run)

Below is using Stable diffusion image generate time

**AMD Radeon AI Pro R9700 (Navi48)**
Run 1 (No Cache) : 24.37s 
Run 2 (With Cache) : 3.01s
Run 3 (Steady State) : 0.91s 
Run 4 (Steady State) : 0.91s 

**NVIDIA RTX 4070 Laptop GPU**
Run 1 (No Cache) : 2.73s
Run 2 (With Cache) : 2.58s
Run 3 (Steady State) : 2.13s
Run 4 (Steady State) : ~2.18s

import torch
from diffusers import StableDiffusionPipeline
import time
import platform
def benchmark_sd():
    # 1. Detection Logic
    if torch.cuda.is_available():
        device = "cuda"
        gpu_name = torch.cuda.get_device_name(0)
    else:
        print("No dGPU detected. Ensure your drivers and ROCm/CUDA PyTorch are installed.")
        return
    print(f"Benchmarking Platform: {platform.system()}")
    print(f"Target dGPU: {gpu_name}")
    print(f"Backend: {torch.version.cuda if 'NVIDIA' in gpu_name else torch.version.hip}")
        # 2. Load Model
    # Using float16 for a fair high-performance comparison
    model_id = "runwayml/stable-diffusion-v1-5"
    pipe = StableDiffusionPipeline.from_pretrained(
        model_id, 
        torch_dtype=torch.float16,
        safety_checker=None
    ).to(device)
    prompt = "A high-tech laboratory with AMD and NVIDIA GPUs, digital art, 4k"
    # 3. Execution Loop
    for i in range(1, 4):
        print(f"\nStarting Run {i}...")
        # Start Timing
        start_time = time.time()
        # Generate Image
        # Note: manual_seed ensures the workload is identical
        generator = torch.Generator(device=device).manual_seed(42)
        image = pipe(prompt, num_inference_steps=20, generator=generator).images[0]
        # Ensure GPU is finished before stopping clock
        if device == "cuda":
            torch.cuda.synchronize()
        end_time = time.time()
        elapsed = end_time - start_time
        print(f"Run {i} Time: {elapsed:.2f} seconds")
        if i == 1:
            print("^^^ (This includes cold-start compilation/allocation) ^^^")
if __name__ == "__main__":
    benchmark_sd()
